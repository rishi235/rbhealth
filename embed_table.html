async function loadTable() { const proxyUrl = 'https://cors-anywhere.herokuapp.com/'; const targetUrl = 'https://rbclinicinfo.weebly.com/'; try { console.log('Fetching data from', targetUrl); const response = await fetch(proxyUrl + targetUrl); if (!response.ok) { throw new Error('Network response was not ok'); } console.log('Response received, parsing HTML...'); const text = await response.text(); console.log('Fetched HTML:', text); // Log the fetched HTML for inspection const parser = new DOMParser(); const doc = parser.parseFromString(text, 'text/html'); console.log('Looking for table wrapper...'); const tableWrapper = doc.querySelector('.simple-table-wrapper'); console.log('Table wrapper found:', tableWrapper); const tableContainer = document.getElementById('table-container'); if (tableWrapper) { console.log('Table found, updating container...'); tableContainer.innerHTML = tableWrapper.outerHTML; } else { console.log('Table not found.'); tableContainer.innerHTML = '

Table not found.

'; } } catch (error) { console.error('Error loading the table:', error); document.getElementById('table-container').innerHTML = `

Error loading the table: ${error.message}

`; } } // Call the function to load the table when the script runs loadTable();
