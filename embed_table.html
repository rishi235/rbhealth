async function loadTable() { const proxyUrl = 'https://cors-anywhere.herokuapp.com/'; const targetUrl = 'https://rbclinicinfo.weebly.com/'; try { const response = await fetch(proxyUrl + targetUrl); if (!response.ok) { throw new Error('Network response was not ok'); } const text = await response.text(); const parser = new DOMParser(); const doc = parser.parseFromString(text, 'text/html'); const tableWrapper = doc.querySelector('.simple-table-wrapper'); const tableContainer = document.getElementById('table-container'); if (tableWrapper) { tableContainer.innerHTML = tableWrapper.outerHTML; } else { tableContainer.innerHTML = '

Table not found.

'; } } catch (error) { console.error('Error loading the table:', error); document.getElementById('table-container').innerHTML = `

Error loading the table: ${error.message}

`; } } // Call the function to load the table when the script runs loadTable();
